#include <bits/stdc++.h>
using namespace std;

int coinChange(int coins[], int numCoins, int amount) {
    int dp[numCoins][amount+1];

    for (int i = 0; i < numCoins; i++) {
        dp[i][0] = 1;
    }

    for (int i = 0; i < numCoins; i++) {
        for (int j = 1; j <= amount; j++) {
            dp[i][j] = (i >= 1) ? dp[i-1][j] : 0;
            if (coins[i] <= j) {
                dp[i][j] += dp[i][j-coins[i]];
            }
        }
    }

    return dp[numCoins-1][amount];
}

int main() {
    int coins[] = {1,3,5};
    int numCoins = 3;
    int amount = 7;
    cout << "Number of ways to make change: " << coinChange(coins, numCoins, amount) << endl;
    return 0;
}
